# Expected 422:
# { "errors": ["Serial number must match format XXX-NNN (e.g. LAP-001)"] }

  -H "Content-Type: application/json" \
  -d '{"equipment": {"name": "Test Laptop", "serial_number": "LAP001", "status": "available", "category_id": 1}}' | jq
# Expected 422: same error

# Rule 2 — performed_at cannot be in the future

  -H "Content-Type: application/json" \
  -d '{"maintenance_record": {"description": "Future service", "performed_at": "2099-01-01T00:00:00", "equipment_id": 1}}' | jq
# Expected 422:
# { "errors": ["Performed at cannot be in the future"] }

# Rule 3 — category name too short (under 3 characters)
curl -s -X POST http://localhost:3000/categories \
  -H "Content-Type: application/json" \
  -d '{"category": {"name": "AB"}}' | jq
# Expected 422:
# { "errors": ["Name is too short (minimum is 3 characters)"] }

# Rule 3 — single character name
curl -s -X POST http://localhost:3000/categories \
  -H "Content-Type: application/json" \
  -d '{"category": {"name": "A"}}' | jq
# Expected 422: same error

# Rule 4 — equipment name all numbers (no letters)
